fait un mode nuit (couleures actuelles) et un mode jour (plutot couleur blanche)
et fait en sorte que quand je clique sur les boutons mixer ça redirige vers une page loading et après 5 sec ça aille sur la page mixer.html
je te met le code pour la page loading pour te donner une idée :
<script>
  const percentageText = document.getElementById('percentage');
  const progressFill = document.getElementById('progress-fill');
  const progressBlocks = document.getElementById('progress-blocks');

  const blocksCount = 20;

  for (let i = 0; i < blocksCount; i++) {
    const block = document.createElement('span');
    progressBlocks.appendChild(block);
  }

  let percent = 0;

  const interval = setInterval(() => {
    if (percent > 100) {
      clearInterval(interval);
      // Redirection après 5 secondes
      window.location.href = 'home.html'; // Remplace par ton URL de destination
      return;
    }

    percentageText.textContent = `${percent}%`;
    progressFill.style.width = `${percent}%`;

    const blocks = progressBlocks.querySelectorAll('span');
    const filledCount = Math.floor((percent / 100) * blocksCount);
    blocks.forEach((block, i) => {
      block.classList.toggle('filled', i < filledCount);
    });

    percent++;
  }, 50); // 50ms * 100 = 5000ms = 5s
</script>
